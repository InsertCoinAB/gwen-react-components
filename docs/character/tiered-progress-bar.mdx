import { Playground, Props } from "docz"
import { TieredProgressBarComponent } from "../../src"
import { Tiers1, Tiers2 } from "./data"

# Tiered Progress Bar Component

Use this component to display detailed character progression

<Playground>
	<div style={{ display: "flex", flexDirection: "column", height: "200px", width: "400px", justifyContent: "space-around" }}>
		<TieredProgressBarComponent stepNumber={35} tiers={Tiers1} icon="circle" color="#13b06d" background="#87c5aa" />
		<TieredProgressBarComponent stepNumber={45} tiers={Tiers2} icon="star" color="#a57af7" background="#bbabd8" />
	</div>
</Playground>

<Playground>
	{() => {
		const [state, setState] = React.useState({ step: 1, theme: { icon: "circle", primary: "#5dca7a", background: "#d5faee" } })
		const themes = [
			{ icon: "circle", primary: "#5dca7a", background: "#d5faee" },
			{ icon: "triangle", primary: "#2583bc", background: "#d6effe" },
			{ icon: "square", primary: "#ebd10a", background: "#fffad9" },
			{ icon: "star", primary: "#ae72fb", background: "#f4edfe" },
		]
		return (
			<React.Fragment>
				<button
					onClick={() =>
						setState((s) => {
							const p = Math.random()
							return { step: s.step + Math.floor(p * 5), theme: themes[Math.floor(p * 4)] }
						})
					}
				>
					Gain Experience
				</button>
				<TieredProgressBarComponent
					stepNumber={state.step}
					tiers={Tiers1}
					icon={state.theme.icon}
					color={state.theme.primary}
					background={state.theme.background}
				/>
			</React.Fragment>
		)
	}}
</Playground>
