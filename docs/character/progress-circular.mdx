<Playground>
	{() => {
		const [state, setState] = React.useState({ step: 1, progress: 0 })
		return (
			<React.Fragment>
				<button
					onClick={() =>
						setState((s) => {
							const p = Math.random()
							return { progress: p, step: p < s.progress ? s.step + 1 : s.step }
						})
					}
				>
					Gain Experience
				</button>
				<ProgressCircleComponent
					step={state.step}
					shape="shield"
					progress={state.progress}
					avatar="/public/bunny.svg"
					stroke={{ primary: "#a57af7" }}
					style={{ marginBottom: "50px" }}
				/>
			</React.Fragment>
		)
	}}
</Playground>
